\section{Definition}

\begin{definition}[Evolution equation]
    \[
        \begin{aligned}
            x : \T &\to \Omega
            t &\ra x(t)
        \end{aligned}            
    \]
\end{definition}

$\T = \RR (\RR^+)$ or $\T = \ZZ (\NN)$.

$\Omega$ = state space.

\begin{definition}[Observation]
    \[
        \begin{aligned}
            y : \T &\to \Theta
            t &\ra y(t)
        \end{aligned}            
    \]
\end{definition}

$\Theta$ = output space.

\begin{definition}[Trajectory]
    \[
        \set{ (t, x(t)) \vert t\in \Omega } \subseteq \T \times \Omega
    \]
\end{definition}

\begin{definition}[Orbit]
    \[
        \set{ t \vert t\in \Omega } \subseteq \Omega
    \]
\end{definition}
